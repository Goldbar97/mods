module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkFishingNet
	{
		Weight			= 2.5,
		Type			= Normal,
		DisplayName		= 10 x Fishing Net Trap,
		Icon			= PFishTrap,
	}
	
	item 5pkFishingNet
	{
		Weight			= 1.75,
		Type			= Normal,
		DisplayName		= 5 x Fishing Net Trap,
		Icon			= PFishTrap,
	}

	recipe Pack 10
	{
		destroy FishingNet=10,
		
		Result:10pkFishingNet,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:100.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkFishingNet,
		
		Result:FishingNet=10,
		Time:75.0,
		Category:Storage,
	}
	
	recipe Pack 5
	{
		destroy FishingNet=5,
		
		Result:5pkFishingNet,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:75.0,
		Category:Storage,
	}
	
	recipe Unpack 5
	{
		destroy 5pkFishingNet,
		
		Result:FishingNet=5,
		Time:50.0,
		Category:Storage,
	}

	recipe Put Together
	{
		5pkFishingNet=2,
		
		Result:10pkFishingNet=1,
		Time:40.0,
	}

	recipe Split in Two
	{
		10pkFishingNet=1,
		
		Result:5pkFishingNet=2,
		Time:40.0,
	}
}
