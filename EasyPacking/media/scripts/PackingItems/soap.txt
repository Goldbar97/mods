module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkSoap
	{
		Weight			= 0.5,
		Type			= Normal,
		DisplayName		= 10 x Soap,
		Icon			= PSoap,
	}
	
	item 5pkSoap
	{
		Weight			= 0.35,
		Type			= Normal,
		DisplayName		= 5 x Soap,
		Icon			= PSoap,
	}

	recipe Pack 10
	{
		destroy Soap2=10,
		
		Result:10pkSoap,
		OnCreate:Recipe.OnCreate.SaveUses,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:75.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkSoap,
		
		Result:Soap2,
		OnCreate:Recipe.OnCreate.LoadUses,
		RemoveResultItem:true,
		Time:50.0,
		Category:Storage,
	}
	
	recipe Pack 5
	{
		destroy Soap2=5,
		
		Result:5pkSoap,
		OnCreate:Recipe.OnCreate.SaveUses,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:50.0,
		Category:Storage,
	}
	
	recipe Unpack 5
	{
		destroy 5pkSoap,
		
		Result:Soap2,
		OnCreate:Recipe.OnCreate.LoadUses,
		RemoveResultItem:true,
		Time:25.0,
		Category:Storage,
	}

	recipe Put Together
	{
		5pkSoap=2,
		
		Result:10pkSoap=1,
		OnCreate:Recipe.OnCreate.MergeUses,
		Time:30.0,
	}

	recipe Split in Two
	{
		10pkSoap=1,
		
		Result:5pkSoap,
		OnCreate:Recipe.OnCreate.SplitUsesInTwo,
		RemoveResultItem:true,
		Time:30.0,
	}
}
