module Packing
{
	imports
	{
		Base,
	}
	
	item 10pkBattery
	{
		Weight			= 0.5,
		Type			= Normal,
		DisplayName		= 10 x Battery,
		Icon			= PBattery,
		WorldStaticModel= PackOfBatteries
	}
	
	item 5pkBattery
	{
		Weight			= 0.35,
		Type			= Normal,
		DisplayName		= 5 x Battery,
		Icon			= PBattery,
		WorldStaticModel= PackOfBatteries
	}

	recipe Pack 10
	{
		destroy Battery=10,
		
		Result:10pkBattery,
		OnCreate:Recipe.OnCreate.SaveUses,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:100.0,
		Category:Storage,
	}
	
	recipe Unpack 10
	{
		destroy 10pkBattery,
		
		Result:Battery,
		OnCreate:Recipe.OnCreate.LoadUses,
		RemoveResultItem:true,
		Time:75.0,
		Category:Storage,
	}
	
	recipe Pack 5
	{
		destroy Battery=5,
		
		Result:5pkBattery,
		OnCreate:Recipe.OnCreate.SaveUses,
		OnTest:Recipe.OnTest.IsFavorite,
		Time:75.0,
		Category:Storage,
	}
	
	recipe Unpack 5
	{
		destroy 5pkBattery,
		
		Result:Battery,
		OnCreate:Recipe.OnCreate.LoadUses,
		RemoveResultItem:true,
		Time:50.0,
		Category:Storage,
	}
	
	recipe Put Together
	{
		5pkBattery=2,
		
		Result:10pkBattery=1,
		OnCreate:Recipe.OnCreate.MergeUses,
		Time:30.0,
	}

	recipe Split in Two
	{
		10pkBattery=1,
		
		Result:5pkBattery,
		OnCreate:Recipe.OnCreate.SplitUsesInTwo,
		RemoveResultItem:true,
		Time:30.0,
	}

	model PackOfBatteries
	{
		mesh = WorldItems/pack_of_batteries,
		texture = WorldItems/PackOfBattery,
		scale = 1.0,
	}
}
